/** * Baseball Stats Automator - Robust Version * データのズレや空行によるバグを防止した修正版 */ const CONFIG = { MAIN_SHEET: "シート1", DATA_SHEET: "フォームの回答 3", START_ROW: 2 }; function updateStats() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const mainSheet = ss.getSheetByName(CONFIG.MAIN_SHEET); const dataSheet = ss.getSheetByName(CONFIG.DATA_SHEET); if (!mainSheet || !dataSheet) { console.error("シート名が見つかりません。設定を確認してください。"); return; } const lastRow = mainSheet.getLastRow(); if (lastRow < CONFIG.START_ROW) return; // 10列分（C列〜L列）の範囲を取得 const targetRange = mainSheet.getRange(CONFIG.START_ROW, 3, lastRow - (CONFIG.START_ROW - 1), 10); const formulas = []; for (let i = CONFIG.START_ROW; i <= lastRow; i++) { formulas.push(buildFormulas(i)); } targetRange.setFormulas(formulas); // 書式設定 mainSheet.getRange(CONFIG.START_ROW, 4, lastRow - 1, 1).setNumberFormat("0.000"); // D列 mainSheet.getRange(CONFIG.START_ROW, 9, lastRow - 1, 1).setNumberFormat("0.000"); // I列 } function buildFormulas(row) { const d = CONFIG.DATA_SHEET; // B列の名前(名前)とM列の役割(投手/野手)を基準にする const name = `$B${row}`; const role = `$M${row}`; // SUMIFの参照先を「フォームの回答 3」の実際の列に固定 const sumif = (col) => `=IFERROR(SUMIF('${d}'!$B:$B, ${name}, '${d}'!${col}), 0)`; return [ sumif("$E:$E"), // C: 試合数 `=IF(${role}="投手", IFERROR($H${row}*9/($J${row}/3), 0), IFERROR($G${row}/$J${row}, 0))`, // D: 打率/防御率 sumif("$F:$F"), // E: 本塁打 sumif("$G:$G"), // F: 打点 sumif("$D:$D"), // G: 安打 sumif("$J:$J"), // H: 長打 `=IF(${role}="投手", IFERROR(($G${row}+$L${row})/($J${row}/3), 0), IFERROR(($G${row}+$L${row}+$H${row}*2)/$J${row}, 0))`, // I: OPS/WHIP sumif("$C:$C"), // J: 打数 sumif("$H:$H"), // K: 盗塁 sumif("$I:$I") // L: 四死球 ]; } function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const triggers = ScriptApp.getProjectTriggers(); triggers.forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('updateStats') .forSpreadsheet(ss) .onFormSubmit() .create(); updateStats(); }
