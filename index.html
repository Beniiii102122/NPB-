/** * Baseball Stats Automator * GitHub: [Your Repository Name] * * 概要: Google Formからの回答を集計し、投手・野手別に統計(打率/防御率等)を算出します。 */ // --- 設定エリア --- const CONFIG = { MAIN_SHEET_NAME: "シート1", // 選手名鑑のメインシート RESPONSE_SHEET_NAME: "フォームの回答 3", // フォームの回答が蓄積されるシート START_ROW: 2, // データの開始行 COLS: { NAME: "B", // 選手名 GAMES: "C", // 試合数/登板数 RATIO: "D", // 打率/防御率 HR_W: "E", // 本塁打/勝利 RBI_L: "F", // 打点/敗戦 HITS_K: "G", // 安打/奪三振 EXTRA_ER: "H", // 長打/自責点 OPS_WHIP: "I", // OPS/WHIP AB_OUTS: "J", // 打数/アウト数 SB_H: "K", // 盗塁/ホールド BB_S: "L", // 四死球/セーブ ROLE: "M" // 投手/野手の判定 } }; /** * フォーム送信時に実行されるメイン処理 */ function onFormSubmitTrigger() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.MAIN_SHEET_NAME); const lastRow = sheet.getLastRow(); if (lastRow < CONFIG.START_ROW) return; const targetRange = sheet.getRange(CONFIG.START_ROW, 3, lastRow - CONFIG.START_ROW + 1, 10); const formulas = []; for (let i = CONFIG.START_ROW; i <= lastRow; i++) { formulas.push(generateRowFormulas(i)); } targetRange.setFormulas(formulas); // 書式設定（小数点3桁） sheet.getRange(`${CONFIG.COLS.RATIO}${CONFIG.START_ROW}:${CONFIG.COLS.RATIO}${lastRow}`).setNumberFormat("0.000"); sheet.getRange(`${CONFIG.COLS.OPS_WHIP}${CONFIG.START_ROW}:${CONFIG.COLS.OPS_WHIP}${lastRow}`).setNumberFormat("0.000"); } /** * 各行の数式を生成するサブ関数 */ function generateRowFormulas(row) { const resp = CONFIG.RESPONSE_SHEET_NAME; const c = CONFIG.COLS; return [ // 試合数 (回答E列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$E:$E), 0)`, // 打率/防御率 `=IF($${c.ROLE}${row}="投手", IFERROR($${c.EXTRA_ER}${row}*9/($${c.AB_OUTS}${row}/3), 0), IFERROR($${c.HITS_K}${row}/$${c.AB_OUTS}${row}, 0))`, // 本塁打/勝利 (回答F列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$F:$F), 0)`, // 打点/敗戦 (回答G列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$G:$G), 0)`, // 安打/奪三振 (回答D列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$D:$D), 0)`, // 長打/自責点 (回答J列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$J:$J), 0)`, // OPS/WHIP `=IF($${c.ROLE}${row}="投手", IFERROR(($${c.HITS_K}${row}+$${c.BB_S}${row})/($${c.AB_OUTS}${row}/3), 0), IFERROR(($${c.HITS_K}${row}+$${c.BB_S}${row}+$${c.EXTRA_ER}${row}*2)/$${c.AB_OUTS}${row}, 0))`, // 打数/アウト数 (回答C列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$C:$C), 0)`, // 盗塁/ホールド (回答H列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$H:$H), 0)`, // 四死球/セーブ (回答I列) `=IFERROR(SUMIF('${resp}'!$B:$B, $${c.NAME}${row}, '${resp}'!$I:$I), 0)` ]; } /** * 初期設定用：トリガー作成 */ function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const triggers = ScriptApp.getProjectTriggers(); triggers.forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('onFormSubmitTrigger') .forSpreadsheet(ss) .onFormSubmit() .create(); onFormSubmitTrigger(); // 初回計算 }
