/** * Baseball Statistics Automator v1.0.0 * * [概要] * Google Formから送信された野球の試合データを集計し、 * 投手・野手それぞれの指標（打率、防御率、OPS、WHIP等）を自動算出します。 */ // ========================================== // 1. 設定：環境に合わせてここを書き換えてください // ========================================== const CONFIG = { SHEET_NAME_MAIN: "シート1", // 表示用メインシート名 SHEET_NAME_RESPONSES: "フォームの回答 3", // 回答蓄積用シート名 // メインシートの列定義（1から始まる列番号） COL: { NAME: 2, // B列: 選手名 GAMES: 3, // C列: 試合数 AVG_ERA: 4, // D列: 打率/防御率 HR_W: 5, // E列: 本塁打/勝利 RBI_L: 6, // F列: 打点/敗戦 HITS_K: 7, // G列: 安打/奪三振 EXTRA_ER: 8, // H列: 長打/自責点 OPS_WHIP: 9, // I列: OPS/WHIP AB_OUTS: 10, // J列: 打数/アウト数 SB_H: 11, // K列: 盗塁/ホールド BB_S: 12, // L列: 四死球/セーブ ROLE: 13 // M列: 役割(投手/野手) } }; // ========================================== // 2. メイン処理：自動更新エンジン // ========================================== function updateStats() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const mainSheet = ss.getSheetByName(CONFIG.SHEET_NAME_MAIN); const lastRow = mainSheet.getLastRow(); if (lastRow < 2) return; // データがない場合は終了 // 計算範囲を指定（C列〜L列） const targetRange = mainSheet.getRange(2, 3, lastRow - 1, 10); const formulas = []; for (let i = 2; i <= lastRow; i++) { formulas.push(buildRowFormulas(i)); } // 数式を一括書き込み（高速化） targetRange.setFormulas(formulas); // 数値フォーマットの適用（小数点以下3桁） applyFormat(mainSheet, lastRow); } /** * 3. 行ごとの数式構築 */ function buildRowFormulas(row) { const r = CONFIG.SHEET_NAME_RESPONSES; const c = CONFIG.COL; const colLetter = (num) => String.fromCharCode(64 + num); const name = `$${colLetter(c.NAME)}${row}`; const role = `$${colLetter(c.ROLE)}${row}`; const sumif = (col) => `=IFERROR(SUMIF('${r}'!$B:$B, ${name}, '${r}'!${col}), 0)`; return [ sumif("$E:$E"), // C: 試合数 `=IF(${role}="投手", IFERROR($${colLetter(c.EXTRA_ER)}${row}*9/($${colLetter(c.AB_OUTS)}${row}/3), 0), IFERROR($${colLetter(c.HITS_K)}${row}/$${colLetter(c.AB_OUTS)}${row}, 0))`, // D: 率 sumif("$F:$F"), // E: 本塁打 sumif("$G:$G"), // F: 打点 sumif("$D:$D"), // G: 安打 sumif("$J:$J"), // H: 長打 `=IF(${role}="投手", IFERROR(($${colLetter(c.HITS_K)}${row}+$${colLetter(c.BB_S)}${row})/($${colLetter(c.AB_OUTS)}${row}/3), 0), IFERROR(($${colLetter(c.HITS_K)}${row}+$${colLetter(c.BB_S)}${row}+$${colLetter(c.EXTRA_ER)}${row}*2)/$${colLetter(c.AB_OUTS)}${row}, 0))`, // I: OPS/WHIP sumif("$C:$C"), // J: 打数 sumif("$H:$H"), // K: 盗塁 sumif("$I:$I") // L: 四死球 ]; } /** * 4. 書式設定 */ function applyFormat(sheet, lastRow) { const c = CONFIG.COL; // 率とOPSの列を小数点3桁表示に sheet.getRange(2, c.AVG_ERA, lastRow - 1, 1).setNumberFormat("0.000"); sheet.getRange(2, c.OPS_WHIP, lastRow - 1, 1).setNumberFormat("0.000"); } /** * 5. 初期化：トリガー設定 * インストール後に一度だけ実行してください */ function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('updateStats') .forSpreadsheet(ss) .onFormSubmit() .create(); updateStats(); }
